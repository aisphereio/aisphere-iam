# Grant Resource Selector Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the access assignment raw resource ID field with a zone-aware selector that only submits registered IAM catalog resources.

**Architecture:** The backend adds proto-first `query` support to `ListResources` and keeps zone/type scoping authoritative. The frontend uses the header `identityOrg` as the only zone context, auto-resolves `zone` grants to that value, and uses a searchable catalog combobox for all other grantable resource types.

**Tech Stack:** Go, protobuf, buf/Kratos-style generated services, React 19, Next.js, TanStack Query, Orval, Vitest, Radix Popover, cmdk.

func (r *MemoryControlPlaneRepository) ListResources(ctx context.Context, opts ListOptions) (*Page[ResourceModel], error) {
	r.mu.RLock()
	defer r.mu.RUnlock()
	var out []ResourceModel
	for _, v := range r.resources {
		searchable := strings.Join([]string{v.ID, v.Slug, v.DisplayName, v.Path}, " ")
		if (opts.Type == "" || v.Type == opts.Type) &&
			(opts.OrgID == "" || v.OrgID == opts.OrgID) &&
			(opts.ProjectID == "" || v.ProjectID == opts.ProjectID) &&
			statusOK(v.Status, opts.Status) &&
			containsFold(searchable, opts.Q) {
			out = append(out, *clone(v))
		}
	}
	return pageOf(out, opts.Page, opts.Size), nil
}

```tsx
'use client';

import { Check, ChevronsUpDown, Loader2, Search } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { Command, CommandEmpty, CommandGroup, CommandInput, CommandItem, CommandList } from '@/components/ui/command';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { cn } from '@/lib/utils';
import type { IamResource } from '@/lib/api/types';

interface ResourceComboboxProps {
  value: string;
  resources: IamResource[];
  open: boolean;
  disabled?: boolean;
  loading?: boolean;
  error?: boolean;
  placeholder?: string;
  search: string;
  onOpenChange: (open: boolean) => void;
  onSearchChange: (value: string) => void;
  onValueChange: (resourceId: string) => void;
}

export function ResourceCombobox({
  value,
  resources,
  open,
  disabled,
  loading,
  error,
  placeholder = '选择资源',
  search,
  onOpenChange,
  onSearchChange,
  onValueChange,
}: ResourceComboboxProps) {
  const selected = resources.find((resource) => resource.ref.id === value);
  return (
    <Popover open={open} onOpenChange={onOpenChange}>
      <PopoverTrigger asChild>
        <Button
          type="button"
          variant="outline"
          role="combobox"
          aria-expanded={open}
          disabled={disabled}
          className="h-11 w-full justify-between px-3 text-left font-normal"
        >
          <span className="min-w-0 truncate">{selected ? resourcePrimaryLabel(selected) : placeholder}</span>
          {loading ? <Loader2 className="ml-2 h-4 w-4 shrink-0 animate-spin opacity-60" /> : <ChevronsUpDown className="ml-2 h-4 w-4 shrink-0 opacity-50" />}
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-[--radix-popover-trigger-width] p-0" align="start">
        <Command shouldFilter={false}>
          <CommandInput value={search} onValueChange={onSearchChange} placeholder="搜索名称、slug、路径或 ID" />
          <CommandList>
            {loading && (
              <div className="flex items-center gap-2 px-3 py-6 text-sm text-muted-foreground">
                <Loader2 className="h-4 w-4 animate-spin" />
                正在加载资源
              </div>
            )}
            {!loading && error && (
              <div className="px-3 py-6 text-sm text-destructive">资源列表加载失败，请重试。</div>
            )}
            {!loading && !error && resources.length === 0 && (
              <CommandEmpty>
                <span className="inline-flex items-center gap-2">
                  <Search className="h-4 w-4 opacity-60" />
                  没有匹配的已注册资源
                </span>
              </CommandEmpty>
            )}
            {!loading && !error && resources.length > 0 && (
              <CommandGroup>
                {resources.map((resource) => (
                  <CommandItem
                    key={`${resource.ref.type}:${resource.ref.id}`}
                    value={resource.ref.id}
                    onSelect={() => {
                      onValueChange(resource.ref.id);
                      onOpenChange(false);
                    }}
                    className="items-start"
                  >
                    <Check className={cn('mt-0.5 h-4 w-4', value === resource.ref.id ? 'opacity-100' : 'opacity-0')} />
                    <span className="min-w-0 flex-1">
                      <span className="block truncate font-medium">{resourcePrimaryLabel(resource)}</span>
                      <span className="block truncate text-xs text-muted-foreground">{resourceSecondaryLabel(resource)}</span>
                    </span>
                  </CommandItem>
                ))}
              </CommandGroup>
            )}
          </CommandList>
        </Command>
      </PopoverContent>
    </Popover>
  );
}

function resourcePrimaryLabel(resource: IamResource): string {
  return resource.displayName || resource.slug || resource.ref.id;
}

function resourceSecondaryLabel(resource: IamResource): string {
  return [resource.slug, resource.path, resource.ref.id].filter(Boolean).join(' / ');
}
```
